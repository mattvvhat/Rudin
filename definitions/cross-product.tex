\begin{equation}
  \vec{a} \times \vec{b} = \vec{c} =
  \left| \begin{array}{ccc}
    \vec{i}   & \vec{j}   & \vec{k}   \\
    a_0       & a_1       & a_2       \\
    b_0       & b_1       & b_2
  \end{array} \right|
=
  \left[ \begin{array}{ccc}
    a_1 b_2 - a_2 b_1     \\
    a_2 b_0 - a_0 b_2    \\
    a_0 b_1 - a_1 b_0
  \end{array} \right]
\end{equation}
