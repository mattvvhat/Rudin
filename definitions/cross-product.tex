\newcommand{\mat}[4]{
  \left| \begin{array}{cc}
    #1 & #2 \\
    #3 & #4
  \end{array} \right|
}

\begin{document}

$ i, j, k \in \mathbb{R}^3 $


\begin{align*}
  \vec{a} \times \vec{b}  & = \vec{c}\\
  & =
  \left| \begin{array}{ccc}
    \vec{i}   & \vec{j}   & \vec{k}   \\
    a_0       & a_1       & a_2       \\
    b_0       & b_1       & b_2
  \end{array} \right| \\
  & = \vec{i} \mat{a_1}{a_2}{b_1}{b_2} - \vec{j} \mat{a_0}{a_2}{b_0}{b_2} + \vec{k}\mat{a_0}{a_1}{b_0}{b_1} \\
  & = 
  \left[ \begin{array}{ccc}
    a_1 b_2 - a_2 b_1     \\
    a_2 b_0 - a_0 b_2    \\
    a_0 b_1 - a_1 b_0
  \end{array} \right]
\end{align*}
